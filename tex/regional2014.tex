\newcommand{\num}[2]{%
    \begin{tikzpicture}[color=black, line width=0.2mm]
      \draw (0,0)
            node {\zsavepos{#1.#2-ll}}
            rectangle (3mm,3mm)
            node [midway, midway] {\small #2}
            node {\zsavepos{#1.#2-ur}};
    \end{tikzpicture}%
    \\
    \write\mywrite{#1.#2 \zposx{#1.#2-ll} \zposy{#1.#2-ll} \zposx{#1.#2-ur} \zposy{#1.#2-ur}}%
}
